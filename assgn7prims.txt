#include <bits/stdc++.h>
using namespace std;

class tree{
    int a[20][20],l,u,w,i,j,branch,e,visited[20];
    public:
    void input();
    void display();
    void minimum();
};

void tree::input(){
    cout<< "\nEnter the no.of branches: ";
    cin >> branch;

    for(i=0;i<branch;i++){
        visited[i] = 0;
        for(j=0;j<branch;j++){
            a[i][j] = 999;
        }
    }

    cout << "Enter the no. of connections: ";
    cin >> e;
    for(i =0;i<e;i++){
        cout << "Enter the end of branches of connections: " << endl;
        cin >> l >> u;
        cout << "Enter the phone company charges of this connections: ";
        cin >> w;
        a[l-1][u-1] = a[u-1][l-1] = w;
    }
}

void tree::display(){
    cout << "\nAdjecency matrix: ";
    for(i=0;i<branch;i++){
        cout << endl;
        for(j=0;j<branch;j++){
            cout << a[i][j]<< " ";
        }
        cout << endl;
    }
}

void tree::minimum(){
    int p=0,q=0,total=0,min=999;
    for(i=0;i<branch;i++){
        for(j=0;j<branch;j++){
            if(min > a[i][j]){
                min = a[i][j];
                p=i;
                q=j;
            }
        }
    }
    visited[p] = 1;
    visited[q] = 1;
    total +=min;
    cout<<"Minimum cost connection is"<<(p+1)<<" -> "<<(q+1)<<"  with charge : "<<min<< endl; 

    for(int count=0;count<(branch-2);count++){
        min = 999;
        for(i=0;i<branch;i++){
            if(visited[i]==1){
                for(j=0;j<branch;j++){
                    if((visited[j] != 1) && (min > a[i][j])){
                        min = a[i][j];
                        p=i;
                        q=j;
                    }
                }
            }
        }
        visited[p]=1;
        visited[q]=1;
        total +=min;
        cout<<"Minimum cost connection is"<<(p+1)<<" -> "<<(q+1)<<"  with charge : "<<min<< endl; 
    }
    cout << "The minimum total cost of connection of all branches is: " << total << endl;
}

int main(){
    int ch;
    tree t;
    do{
		cout<<"==========PRIM'S ALGORITHM================="<<endl;
		cout<<"\n1.INPUT\n \n2.DISPLAY\n \n3.MINIMUM\n"<<endl;
		cout<<"Enter your choice :"<<endl;
		cin>>ch;
	
        switch(ch){	
        case 1: cout<<"*******INPUT YOUR VALUES*******"<<endl;	
            t.input();
            break;

        case 2: cout<<"*******DISPLAY THE CONTENTS********"<<endl;
            t.display();
            break;

        case 3: cout<<"*********MINIMUM************"<<endl;
            t.minimum();
            break;
        }    
	}while(ch!=4);
	return 0;
}
